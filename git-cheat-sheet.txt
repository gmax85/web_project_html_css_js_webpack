1. git status - проверка изменений 
2. git checkout '. or file name' - откат изменений во всех файлах либо в конкретном файле по имени 
3. git add '. or file name' - добавить все файлы или файл в дерево изменений
4. git commit -m "текст коммита" - закоммитить изменения равно сохранить их в локальной ветке
5. git push -u origin 'название ветки' - отправка закоммиченных изменений в удаленый репозиторий
6. git branch - показывает локальных веток
7. git checkout 'название ветки' - переход на ветку по названию
8. git checkout -b 'название новой ветки' - создание новой ветки
9. git pull - затягиваем последнее состояние ветки из удаленного репозитория
10. git merge 'название ветки' - мердж веток
